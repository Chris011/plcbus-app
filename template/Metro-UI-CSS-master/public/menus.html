            <ul class="menu">
                <li data-role="dropdown">
                    <a href="#">Item 1</a>
                    <ul class="dropdown-menu">
                        <li><a href="#">SubItem</a></li>
                        ...
                        <li><a href="#">SubItem</a></li>
                    </ul>
                </li>
                <li><a /></li>

                <li><a /></li>
            </ul>
        </div>
    </div>
